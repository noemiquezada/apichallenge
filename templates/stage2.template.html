function needleInAHaystack(needle, haystack) {
    // Search for needle string in haystack 
    for (var i = 0; i < haystack.length; i++) {
        // If needle is found in haystack return index
        if (needle === haystack[i]) {
            return i; 
        }
    }
    // Else return 01
    return -1;
}
